class Calculadora:
    def somar(self, n1 ,n2):
        Result = n1+n2
        return(Result)
    def subtrair(self, n1 , n2):
        Result = n1-n2
        return(Result)
    def multiplicar(self, n1 , n2):
        Result = n1*n2
        return(Result)
    def dividir(self, n1, n2):
        Result = n1/n2
        return(Result)
